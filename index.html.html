// 🔥 FINAL LEVEL HOSTEL MANAGEMENT SYSTEM (ADMIN DASHBOARD + DEFAULTER + MODERN UI + DEPLOY READY) // ================= BACKEND ================= // Install: // npm init -y // npm install express mongoose multer cors xlsx pdfkit const express = require('express'); const mongoose = require('mongoose'); const multer = require('multer'); const XLSX = require('xlsx'); const cors = require('cors'); const PDFDocument = require('pdfkit'); const fs = require('fs'); const app = express(); app.use(cors()); app.use(express.json()); mongoose.connect('mongodb://127.0.0.1:27017/hostelDB'); // Schemas const User = mongoose.model('User', new mongoose.Schema({ email: String, password: String })); const Student = mongoose.model('Student', new mongoose.Schema({}, { strict: false })); const upload = multer({ dest: 'uploads/' }); // Register/Login app.post('/register', async (req, res) => { await User.create(req.body); res.send('Registered'); }); app.post('/login', async (req, res) => { const user = await User.findOne(req.body); res.send(user ? 'Success' : 'Fail'); }); // Upload Excel app.post('/upload', upload.single('file'), async (req, res) => { const workbook = XLSX.readFile(req.file.path); const sheet = workbook.Sheets[workbook.SheetNames[0]]; const data = XLSX.utils.sheet_to_json(sheet); await Student.deleteMany(); await Student.insertMany(data); res.send('Excel Uploaded'); }); // Search app.get('/student/:code', async (req, res) => { const student = await Student.findOne({ code: req.params.code }); res.json(student); }); // Payment app.post('/payment', async (req, res) => { const { code, amount } = req.body; await Student.updateOne({ code }, { $set: { payment: amount } }); res.send('Payment Saved'); }); // Dashboard Data app.get('/dashboard', async (req, res) => { const total = await Student.countDocuments(); const paid = await Student.countDocuments({ payment: { $exists: true } }); const unpaid = await Student.find({ payment: { $exists: false } }); res.json({ total, paid, unpaid }); }); // PDF Receipt app.get('/receipt/:code', async (req, res) => { const student = await Student.findOne({ code: req.params.code }); const doc = new PDFDocument(); const filePath = `receipt_${student.code}.pdf`; doc.pipe(fs.createWriteStream(filePath)); doc.fontSize(20).text('Hostel Receipt'); doc.text(`Name: ${student.name}`); doc.text(`Code: ${student.code}`); doc.text(`Amount: ${student.payment}`); doc.end(); setTimeout(() => res.download(filePath), 1000); }); app.listen(3000, () => console.log('🚀 Server Running on 3000')); // ================= FRONTEND ================= Hostel Final System
